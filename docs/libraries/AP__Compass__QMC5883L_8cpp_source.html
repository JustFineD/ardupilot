90 {

104  }
105 
106  if (!_dev->write_register(0x0B, 0x01)||
107  !_dev->write_register(0x20, 0x40)||
108  !_dev->write_register(0x21, 0x01)||
109  !_dev->write_register(QMC5883L_REG_CONF1,
110  QMC5883L_MODE_CONTINUOUS|

112  QMC5883L_OSR_512|
113  QMC5883L_RNG_8G)) {
114  goto fail;
115  }
116 
117  // lower retries for run
118  _dev->set_retries(3);
119 
120  _dev->get_semaphore()->give();
121 
122  _instance = register_compass();
123

147 void AP_Compass_QMC5883L::timer()
148 {
149  struct PACKED {
150  le16_t rx;
151  le16_t ry;
152  le16_t rz;
153  } buffer;
154 
155  const float range_scale = 1000.0f / 3000.0f;
156 
157  uint8_t status;
158  if(!_dev->read_registers(QMC5883L_REG_STATUS,&status,1)){
159  return;
160  }
161  //new data is ready
162  if (!(status & 0x04)) {
163  return;
164  }
165 
166  if(!_dev->read_registers(QMC5883L_REG_DATA_OUTPUT_X, (uint8_t *) &buffer, sizeof(buffer))){
167  return ;
168  }
169 
170  uint32_t now = AP_HAL::micros();
171 
172  auto x = -static_cast<int16_t>(le16toh(buffer.rx));
173  auto y = static_cast<int16_t>(le16toh(buffer.ry));
174  auto z = -static_cast<int16_t>(le16toh(buffer.rz));
175 
176 #if 0
177  printf("mag.x:%d\n",x);
178  printf("mag.y:%d\n",y);
179  printf("mag.z:%d\n",z);
180 #endif
181 
182  Vector3f field = Vector3f{x * range_scale , y * range_scale, z * range_scale };
183 
184  // rotate to the desired orientation
185  if (is_external(_instance)) {
186  field.rotate(ROTATION_YAW_90);
187  }
188 
189  /* rotate raw_field from sensor frame to body frame */
190  rotate_field(field, _instance);
191 
192  /* publish raw_field (uncorrected point sample) for calibration use */
193  publish_raw_field(field, now, _instance);
194 
195  /* correct raw_field for known errors */
196  correct_field(field, _instance);
197 
198  if (_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
199  _accum += field;
200  _accum_count++;
201  if(_accum_count == 20){
202  _accum.x /= 2;
203  _accum.y /= 2;
204  _accum.z /= 2;
205  _accum_count = 10;
206  }
207  _sem->give();
208  }
209 }
