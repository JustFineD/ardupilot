36 GCS_MAVLINK::queued_param_send()
37 {
38  if (!initialised) {
39  return;
40  }
41 
42  // send one parameter async reply if pending
43  send_parameter_reply();
44 
45  if (_queued_parameter == nullptr) {
46  return;
47  }
48 
49  uint16_t bytes_allowed;
50  uint8_t count;
51  uint32_t tnow = AP_HAL::millis();
52  uint32_t tstart = AP_HAL::micros();
53 
54  // use at most 30% of bandwidth on parameters. The constant 26 is
55  // 1/(1000 * 1/8 * 0.001 * 0.3)
56  bytes_allowed = 57 * (tnow - _queued_parameter_send_time_ms) * 26;
57  if (bytes_allowed > comm_get_txspace(chan)) {
58  bytes_allowed = comm_get_txspace(chan);
59  }
60  count = bytes_allowed / (MAVLINK_MSG_ID_PARAM_VALUE_LEN + packet_overhead());
61 
62  // when we don't have flow control we really need to keep the
63  // param download very slow, or it tends to stall
64  if (!have_flow_control() && count > 5) {
65  count = 5;
66  }
67 
68  while (_queued_parameter != nullptr && count--) {
69  AP_Param *vp;
70  float value;
71 
72  // copy the current parameter and prepare to move to the next
73  vp = _queued_parameter;
74 
75  // if the parameter can be cast to float, report it here and break out of the loop
76  value = vp->cast_to_float(_queued_parameter_type);
77 
78  char param_name[AP_MAX_NAME_SIZE];
79  vp->copy_name_token(_queued_parameter_token, param_name, sizeof(param_name), true);
80 
81  mavlink_msg_param_value_send(
82  chan,
83  param_name,
84  value,
85  mav_var_type(_queued_parameter_type),
86  _queued_parameter_count,
87  _queued_parameter_index);
88 
89  _queued_parameter = AP_Param::next_scalar(&_queued_parameter_token, &_queued_parameter_type);
90  _queued_parameter_index++;
91 
92  if (AP_HAL::micros() - tstart > 1000) {
93  // don't use more than 1ms sending blocks of parameters
94  break;
95  }
96  }
97  _queued_parameter_send_time_ms = tnow;
98 }

103 bool GCS_MAVLINK::have_flow_control(void)
104 {
105  if (!valid_channel(chan)) {
106  return false;
107  }
108 
109  if (mavlink_comm_port[chan] == nullptr) {
110  return false;
111  }
112 
113  if (chan == MAVLINK_COMM_0) {
114  // assume USB console has flow control
115  return hal.gpio->usb_connected() || mavlink_comm_port[chan]->get_flow_control() != AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE;
116  } else {
117  // all other channels
118  return mavlink_comm_port[chan]->get_flow_control() != AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE;
119  }

124  handle a request to change stream rate. Note that copter passes in
125  save==false so we don't want the save to happen when the user connects the
126  ground station.
127  */
128 void GCS_MAVLINK::handle_request_data_stream(mavlink_message_t *msg, bool save)
129 {
130  mavlink_request_data_stream_t packet;
131  mavlink_msg_request_data_stream_decode(msg, &packet);
132 
133  int16_t freq = 0; // packet frequency
134 
135  if (packet.start_stop == 0)
136  freq = 0; // stop sending
137  else if (packet.start_stop == 1)
138  freq = packet.req_message_rate; // start sending
139  else
140  return;
141 
142  AP_Int16 *rate = nullptr;
143  switch (packet.req_stream_id) {
144  case MAV_DATA_STREAM_ALL:
145  // note that we don't set STREAM_PARAMS - that is internal only
146  for (uint8_t i=0; i<STREAM_PARAMS; i++) {
147  if (save) {
148  streamRates[i].set_and_save_ifchanged(freq);
149  } else {
150  streamRates[i].set(freq);
151  }
152  }
153  break;
154  case MAV_DATA_STREAM_RAW_SENSORS:
155  rate = &streamRates[STREAM_RAW_SENSORS];
156  break;
157  case MAV_DATA_STREAM_EXTENDED_STATUS:
158  rate = &streamRates[STREAM_EXTENDED_STATUS];
159  break;
160  case MAV_DATA_STREAM_RC_CHANNELS:
161  rate = &streamRates[STREAM_RC_CHANNELS];
162  break;
163  case MAV_DATA_STREAM_RAW_CONTROLLER:
164  rate = &streamRates[STREAM_RAW_CONTROLLER];
165  break;
166  case MAV_DATA_STREAM_POSITION:
167  rate = &streamRates[STREAM_POSITION];
168  break;
169  case MAV_DATA_STREAM_EXTRA1:
170  rate = &streamRates[STREAM_EXTRA1];
171  break;
172  case MAV_DATA_STREAM_EXTRA2:
173  rate = &streamRates[STREAM_EXTRA2];
174  break;
175  case MAV_DATA_STREAM_EXTRA3:
176  rate = &streamRates[STREAM_EXTRA3];
177  break;
178  }
179 
180  if (rate != nullptr) {
181  if (save) {
182  rate->set_and_save_ifchanged(freq);
183  } else {
184  rate->set(freq);
185  }
186  }
187 }

207 {

238 void GCS_MAVLINK::handle_param_set(mavlink_message_t *msg, DataFlash_Class *DataFlash)
239 {
240  mavlink_param_set_t packet;
241  mavlink_msg_param_set_decode(msg, &packet);
242  enum ap_var_type var_type;
243 
244  // set parameter
245  AP_Param *vp;
246  char key[AP_MAX_NAME_SIZE+1];
247  strncpy(key, (char *)packet.param_id, AP_MAX_NAME_SIZE);
248  key[AP_MAX_NAME_SIZE] = 0;
249 
250  // find existing param so we can get the old value
251  vp = AP_Param::find(key, &var_type);
252  if (vp == nullptr) {
253  return;
254  }
255  float old_value = vp->cast_to_float(var_type);
256 
257  // set the value
258  vp->set_float(packet.param_value, var_type);
259 
260  /*
261  we force the save if the value is not equal to the old
262  value. This copes with the use of override values in
263  constructors, such as PID elements. Otherwise a set to the
264  default value which differs from the constructor value doesn't
265  save the change
266  */
267  bool force_save = !is_equal(packet.param_value, old_value);
268 
269  // save the change
270  vp->save(force_save);
271 
272  if (DataFlash != nullptr) {
273  DataFlash->Log_Write_Parameter(key, vp->cast_to_float(var_type));
274  }
275 }

277 // see if we should send a stream now. Called at 50Hz
278 bool GCS_MAVLINK::stream_trigger(enum streams stream_num)
279 {
280  if (stream_num >= NUM_STREAMS) {
281  return false;
282  }
283  float rate = (uint8_t)streamRates[stream_num].get();
284 
285  rate *= adjust_rate_for_stream_trigger(stream_num);
286 
287  if (rate <= 0) {
288  if (chan_is_streaming & (1U<<(chan-MAVLINK_COMM_0))) {
289  // if currently streaming then check if all streams are disabled
290  // to allow runtime detection of user disabling streaming
291  bool is_streaming = false;
292  for (uint8_t i=0; i<stream_num; i++) {
293  if (streamRates[stream_num] > 0) {
294  is_streaming = true;
295  }
296  }
297  if (!is_streaming) {
298  // all streams have been turned off, clear the bit flag
299  chan_is_streaming &= ~(1U<<(chan-MAVLINK_COMM_0));
300  }
301  }
302  return false;
303  } else {
304  chan_is_streaming |= (1U<<(chan-MAVLINK_COMM_0));
305  }
306 
307  if (stream_ticks[stream_num] == 0) {
308  // we're triggering now, setup the next trigger point
309  if (rate > 50) {
310  rate = 50;
311  }
312  stream_ticks[stream_num] = (50 / rate) - 1 + stream_slowdown;
313  return true;
314  }
315 
316  // count down at 50Hz
317  stream_ticks[stream_num]--;
318  return false;
319 }

373 void GCS_MAVLINK::param_io_timer(void)
374 {
375  struct pending_param_request req;
376 
377  // this is mostly a no-op, but doing this here means we won't
378  // block the main thread counting parameters (~30ms on PH)
379  AP_Param::count_parameters();
380 
381  if (param_replies.space() == 0) {
382  // no room
383  return;
384  }
385 
386  if (!param_requests.pop(req)) {
387  // nothing to do
388  return;
389  }
390 
391  struct pending_param_reply reply;
392  AP_Param *vp;
393 
394  if (req.param_index != -1) {
395  AP_Param::ParamToken token;
396  vp = AP_Param::find_by_index(req.param_index, &reply.p_type, &token);
397  if (vp == nullptr) {
398  return;
399  }
400  vp->copy_name_token(token, reply.param_name, AP_MAX_NAME_SIZE, true);
401  } else {
402  strncpy(reply.param_name, req.param_name, AP_MAX_NAME_SIZE+1);
403  vp = AP_Param::find(req.param_name, &reply.p_type);
404  if (vp == nullptr) {
405  return;
406  }
407  }
408 
409  reply.chan = req.chan;
410  reply.param_name[AP_MAX_NAME_SIZE] = 0;
411  reply.value = vp->cast_to_float(reply.p_type);
412  reply.param_index = req.param_index;
413  reply.count = AP_Param::count_parameters();
414 
415  // queue for transmission
416  param_replies.push(reply);
417 }
