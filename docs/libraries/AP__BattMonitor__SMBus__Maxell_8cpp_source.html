142 bool AP_BattMonitor_SMBus_Maxell::check_pec_support()
143 {
144  // exit immediately if we have already confirmed pec support
145  if (_pec_confirmed) {
146  return true;
147  }
148 
149  // specification info
150  uint16_t data;
151  if (!read_word(BATTMONITOR_SMBUS_MAXELL_SPECIFICATION_INFO, data)) {
152  return false;
153  }
154 
155  // extract version
156  uint8_t version = (data & 0xF0) >> 4;
157 
158  // version less than 0011b (i.e. 3) do not support PEC
159  if (version < 3) {
160  _pec_supported = false;
161  _pec_confirmed = true;
162  return true;
163  }
164 
165  // check manufacturer name
166  uint8_t buff[SMBUS_READ_BLOCK_MAXIMUM_TRANSFER + 1];
167  if (read_block(BATTMONITOR_SMBUS_MAXELL_MANUFACTURE_NAME, buff, true)) {
168  // Hitachi maxell batteries do not support PEC
169  if (strcmp((char*)buff, "Hitachi maxell") == 0) {
170  _pec_supported = false;
171  _pec_confirmed = true;
172  return true;
173  }
174  }
175 
176  // assume all other batteries support PEC
177  _pec_supported = true;
178  _pec_confirmed = true;
179  return true;
