471  /*
472  buffer for re-assembling RTCM data for GPS injection.
473  The 8 bit flags field in GPS_RTCM_DATA is interpreted as:
474  1 bit for "is fragmented"
475  2 bits for fragment number
476  5 bits for sequence number
477 
478  The rtcm_buffer is allocated on first use. Once a block of data
479  is successfully reassembled it is injected into all active GPS
480  backends. This assumes we don't want more than 4*180=720 bytes
481  in a RTCM data block
482  */
483  struct rtcm_buffer {
484  uint8_t fragments_received:4;
485  uint8_t sequence:5;
486  uint8_t fragment_count;
487  uint16_t total_length;
488  uint8_t buffer[MAVLINK_MSG_GPS_RTCM_DATA_FIELD_DATA_LEN*4];
489  } *rtcm_buffer;
