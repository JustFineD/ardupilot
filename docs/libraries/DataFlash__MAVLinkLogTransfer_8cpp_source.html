105 void DataFlash_Class::handle_log_request_data(GCS_MAVLINK &link, mavlink_message_t *msg)
106 {
107  if (_log_sending_chan >= 0) {
108  // some GCS (e.g. MAVProxy) attempt to stream request_data
109  // messages when they're filling gaps in the downloaded logs.
110  // This channel check avoids complaining to them, at the cost
111  // of silently dropping any repeated attempts to start logging
112  if (_log_sending_chan != link.get_chan()) {
113  link.send_text(MAV_SEVERITY_INFO, "Log download in progress");
114  }
115  return;
116  }
117 
118  mavlink_log_request_data_t packet;
119  mavlink_msg_log_request_data_decode(msg, &packet);
120 
121  _in_log_download = true;
122 
123  _log_listing = false;
124  if (!_log_sending || _log_num_data != packet.id) {
125  _log_sending = false;
126 
127  uint16_t num_logs = get_num_logs();
128  if (packet.id > num_logs || packet.id < 1) {
129  return;
130  }
131 
132  uint32_t time_utc, size;
133  get_log_info(packet.id, size, time_utc);
134  _log_num_data = packet.id;
135  _log_data_size = size;
136 
137  uint16_t end;
138  get_log_boundaries(packet.id, _log_data_page, end);
139  }
140 
141  _log_data_offset = packet.ofs;
142  if (_log_data_offset >= _log_data_size) {
143  _log_data_remaining = 0;
144  } else {
145  _log_data_remaining = _log_data_size - _log_data_offset;
146  }
147  if (_log_data_remaining > packet.count) {
148  _log_data_remaining = packet.count;
149  }
150  _log_sending = true;
151  _log_sending_chan = link.get_chan();
152 
153  handle_log_send(link);
154 }

182 void DataFlash_Class::handle_log_send(GCS_MAVLINK &link)
183 {
184  if (_log_sending_chan != link.get_chan()) {
185  return;
186  }
187  if (_log_listing) {
188  handle_log_send_listing(link);
189  }
190  if (!_log_sending) {
191  return;
192  }
193 
194 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
195  // assume USB speeds in SITL for the purposes of log download
196  const uint8_t num_sends = 40;
197 #else
198  uint8_t num_sends = 1;
199  if (link.is_high_bandwidth() && hal.gpio->usb_connected()) {
200  // when on USB we can send a lot more data
201  num_sends = 250;
202  } else if (link.have_flow_control()) {
203  #if CONFIG_HAL_BOARD == HAL_BOARD_LINUX
204  num_sends = 80;
205  #else
206  num_sends = 10;
207  #endif
208  }
209 #endif
210 
211  for (uint8_t i=0; i<num_sends; i++) {
212  if (_log_sending) {
213  if (!handle_log_send_data(link)) break;
214  }
215  }
216 }

221 void DataFlash_Class::handle_log_send_listing(GCS_MAVLINK &link)
222 {
223  if (!HAVE_PAYLOAD_SPACE(link.get_chan(), LOG_ENTRY)) {
224  // no space
225  return;
226  }
227  if (AP_HAL::millis() - link.get_last_heartbeat_time() > 3000) {
228  // give a heartbeat a chance
229  return;
230  }
231 
232  uint32_t size, time_utc;
233  if (_log_next_list_entry == 0) {
234  size = 0;
235  time_utc = 0;
236  } else {
237  get_log_info(_log_next_list_entry, size, time_utc);
238  }
239  mavlink_msg_log_entry_send(link.get_chan(), _log_next_list_entry, _log_num_logs, _log_last_list_entry, time_utc, size);
240  if (_log_next_list_entry == _log_last_list_entry) {
241  _log_listing = false;
242  _log_sending_chan = -1;
243  } else {
244  _log_next_list_entry++;
245  }
246 }

251 bool DataFlash_Class::handle_log_send_data(GCS_MAVLINK &link)
252 {
253  if (!HAVE_PAYLOAD_SPACE(link.get_chan(), LOG_DATA)) {
254  // no space
255  return false;
256  }
257  if (AP_HAL::millis() - link.get_last_heartbeat_time() > 3000) {
258  // give a heartbeat a chance
259  return false;
260  }
261 
262  int16_t ret = 0;
263  uint32_t len = _log_data_remaining;
264  mavlink_log_data_t packet;
265 
266  if (len > 90) {
267  len = 90;
268  }
269  ret = get_log_data(_log_num_data, _log_data_page, _log_data_offset, len, packet.data);
270  if (ret < 0) {
271  // report as EOF on error
272  ret = 0;
273  }
274  if (ret < 90) {
275  memset(&packet.data[ret], 0, 90-ret);
276  }
277 
278  packet.ofs = _log_data_offset;
279  packet.id = _log_num_data;
280  packet.count = ret;
281  _mav_finalize_message_chan_send(link.get_chan(), MAVLINK_MSG_ID_LOG_DATA, (const char *)&packet,
282  MAVLINK_MSG_ID_LOG_DATA_MIN_LEN,
283  MAVLINK_MSG_ID_LOG_DATA_LEN,
284  MAVLINK_MSG_ID_LOG_DATA_CRC);
285 
286  _log_data_offset += len;
287  _log_data_remaining -= len;
288  if (ret < 90 || _log_data_remaining == 0) {
289  _log_sending = false;
290  _log_sending_chan = -1;
291  }
292  return true;
293 }
