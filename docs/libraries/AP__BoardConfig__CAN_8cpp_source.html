101 void AP_BoardConfig_CAN::px4_setup_canbus(void)
102 {
103  // Create all drivers that we need
104  bool initret = true;
105  for (uint8_t i = 0; i < MAX_NUMBER_OF_CAN_INTERFACES; i++) {
106  uint8_t drv_num = _var_info_can[i]._driver_number;
107 
108  if (drv_num != 0 && drv_num <= MAX_NUMBER_OF_CAN_DRIVERS) {
109  if (hal.can_mgr[drv_num - 1] == nullptr) {
110  const_cast <AP_HAL::HAL&> (hal).can_mgr[drv_num - 1] = new PX4::PX4CANManager;
111  }
112 
113  if (hal.can_mgr[drv_num - 1] != nullptr) {
114  initret &= hal.can_mgr[drv_num - 1]->begin(_var_info_can[i]._can_bitrate, i);
115  } else {
116  printf("Failed to initialize can interface %d\n\r", i + 1);
117  }
118  }
119  }
120 
121  bool any_uavcan_present = false;
122 
123  if (initret) {
124  for (uint8_t i = 0; i < MAX_NUMBER_OF_CAN_DRIVERS; i++) {
125  if (hal.can_mgr[i] != nullptr) {
126  hal.can_mgr[i]->initialized(true);
127  printf("can_mgr %d initialized well\n\r", i + 1);
128 
129  if (_var_info_can_protocol[i]._protocol == UAVCAN_PROTOCOL_ENABLE) {
130  _var_info_can_protocol[i]._uavcan = new AP_UAVCAN;
131 
132  if (_var_info_can_protocol[i]._uavcan != nullptr)
133  {
134  AP_Param::load_object_from_eeprom(_var_info_can_protocol[i]._uavcan, AP_UAVCAN::var_info);
135 
136  hal.can_mgr[i]->set_UAVCAN(_var_info_can_protocol[i]._uavcan);
137  _var_info_can_protocol[i]._uavcan->set_parent_can_mgr(hal.can_mgr[i]);
138 
139  if (_var_info_can_protocol[i]._uavcan->try_init() == true) {
140  any_uavcan_present = true;
141  } else {
142  printf("Failed to initialize uavcan interface %d\n\r", i + 1);
143  }
144  } else {
145  AP_HAL::panic("Failed to allocate uavcan %d\n\r", i + 1);
146  }
147  }
148  }
149  }
150 
151  if (any_uavcan_present) {
152  // start UAVCAN working thread
153  hal.scheduler->create_uavcan_thread();
154 
155  // Delay for magnetometer and barometer discovery
156  hal.scheduler->delay(5000);
157  }
158  }
159 }
