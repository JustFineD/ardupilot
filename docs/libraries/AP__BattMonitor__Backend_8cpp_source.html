48 // update battery resistance estimate
49 // faster rates of change of the current and voltage readings cause faster updates to the resistance estimate
50 // the battery resistance is calculated by comparing the latest current and voltage readings to a low-pass filtered current and voltage
51 // high current steps are integrated into the resistance estimate by varying the time constant of the resistance filter
52 void AP_BattMonitor_Backend::update_resistance_estimate()
53 {
54  // return immediately if no current
55  if (!has_current() || !is_positive(_state.current_amps)) {
56  return;
57  }
58 
59  // update maximum current seen since startup and protect against divide by zero
60  _current_max_amps = MAX(_current_max_amps, _state.current_amps);
61  float current_delta = _state.current_amps - _current_filt_amps;
62  if (is_zero(current_delta)) {
63  return;
64  }
65 
66  // update reference voltage and current
67  if (_state.voltage > _resistance_voltage_ref) {
68  _resistance_voltage_ref = _state.voltage;
69  _resistance_current_ref = _state.current_amps;
70  }
71 
72  // calculate time since last update
73  uint32_t now = AP_HAL::millis();
74  float loop_interval = (now - _resistance_timer_ms) / 1000.0f;
75  _resistance_timer_ms = now;
76 
77  // estimate short-term resistance
78  float filt_alpha = constrain_float(loop_interval/(loop_interval + AP_BATT_MONITOR_RES_EST_TC_1), 0.0f, 0.5f);
79  float resistance_alpha = MIN(1, AP_BATT_MONITOR_RES_EST_TC_2*fabsf((_state.current_amps-_current_filt_amps)/_current_max_amps));
80  float resistance_estimate = -(_state.voltage-_voltage_filt)/current_delta;
81  if (is_positive(resistance_estimate)) {
82  _state.resistance = _state.resistance*(1-resistance_alpha) + resistance_estimate*resistance_alpha;
83  }
84 
85  // calculate maximum resistance
86  if ((_resistance_voltage_ref > _state.voltage) && (_state.current_amps > _resistance_current_ref)) {
87  float resistance_max = (_resistance_voltage_ref - _state.voltage) / (_state.current_amps - _resistance_current_ref);
88  _state.resistance = MIN(_state.resistance, resistance_max);
89  }
90 
91  // update the filtered voltage and currents
92  _voltage_filt = _voltage_filt*(1-filt_alpha) + _state.voltage*filt_alpha;
93  _current_filt_amps = _current_filt_amps*(1-filt_alpha) + _state.current_amps*filt_alpha;
94 
95  // update estimated voltage without sag
96  _state.voltage_resting_estimate = _state.voltage + _state.current_amps * _state.resistance;
97 }
