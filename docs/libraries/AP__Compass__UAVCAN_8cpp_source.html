62  if (AP_BoardConfig_CAN::get_can_num_ifaces() != 0) {
63  for (uint8_t i = 0; i < MAX_NUMBER_OF_CAN_DRIVERS; i++) {

149 void AP_Compass_UAVCAN::handle_mag_msg(Vector3f &mag)
150 {
151  Vector3f raw_field = mag * 1000.0;
152 
153  // rotate raw_field from sensor frame to body frame
154  rotate_field(raw_field, _instance);
155 
156  _last_timestamp = AP_HAL::micros64();
157  // publish raw_field (uncorrected point sample) for calibration use
158  publish_raw_field(raw_field, (uint32_t) _last_timestamp, _instance);
159 
160  // correct raw_field for known errors
161  correct_field(raw_field, _instance);
162 
163  if (_mag_baro->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
164  // accumulate into averaging filter
165  _sum += raw_field;
166  _count++;
167  _mag_baro->give();
168  }
